#!/bin/bash -eu

echo 'Be sure to install brew first'
brew install vim neovim git tmux python3 the_silver_searcher flake8 fortune cowsay lolcat
brew cask install iterm2

# python packages
pip3 install --upgrade neovim

# git setup
if ! [ -f ~/.gitconfig ]; then
    git config --global user.name "Will Deuschle"
    git config --global user.email "wjdeuschle@gmail.com"
fi

# dotfiles setup
if ! [ -d ~/dotfiles ]; then
    git clone https://github.com/willdeuschle/dotfiles.git ~/dotfiles
fi
if ! [ -f ~/.tmux.conf ]; then
    ln -s ~/dotfiles/tmux/.tmux.conf ~/.tmux.conf
    ln -s ~/dotfiles/tmux/.tmuxline.snapshot.txt ~/.tmuxline.snapshot.txt
fi
if ! [ -f ~/.vimrc ]; then
    ln -s ~/dotfiles/vim/vimrc ~/.vimrc
fi
if ! [ -d ~/.config ]; then
    mkdir ~/.config
fi
if ! [ -d ~/.config/nvim ]; then
    mkdir ~/.config/nvim
fi
if ! [ -f ~/.config/nvim/init.vim ]; then
    ln -s ~/dotfiles/vim/vimrc ~/.config/nvim/init.vim
fi

# bashrc
cat ~/dotfiles/.bashrc >> ~/.bashrc

# bash_profile setup
echo "source ~/.bashrc" >> ~/.bash_profile

source ~/.bashrc

vim +'PlugInstall --sync' +qa
