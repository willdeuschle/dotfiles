#!/bin/bash -eu

# apt packages
sudo apt install -y vim neovim wget curl gnome-tweak-tool git tmux python3-pip fonts-powerline silversearcher-ag flake8 htop rsync fortune-mod cowsay lolcat xclip

# install chrome
sudo apt-get install libxss1 libappindicator1 libindicator7
wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb
sudo apt install ./google-chrome*.deb
rm ./google-chrome*.deb

# python packages
pip3 install --upgrade neovim

# git setup
if ! [ -f ~/.gitconfig ]; then
    git config --global user.name "Will Deuschle"
    git config --global user.email "wjdeuschle@gmail.com"
fi

# dotfiles setup
if ! [ -d ~/dotfiles ]; then
    git clone https://github.com/willdeuschle/dotfiles.git ~/dotfiles
fi
if ! [ -f ~/.tmux.conf ]; then
    ln -s ~/dotfiles/tmux/.tmux.conf ~/.tmux.conf
fi
if ! [ -f ~/.vimrc ]; then
    ln -s ~/dotfiles/vim/vimrc ~/.vimrc
fi
if ! [ -d ~/.config ]; then
    mkdir ~/.config
fi
if ! [ -d ~/.config/nvim ]; then
    mkdir ~/.config/nvim
fi
if ! [ -f ~/.config/nvim/init.vim ]; then
    ln -s ~/dotfiles/vim/vimrc ~/.config/nvim/init.vim
fi

# setup keyboard shortcuts
source "$PWD/dotfiles/ubuntu_keybindings.txt"

# bashrc
echo "alias vim='nvim'" >> ~/.bashrc
echo "export EDITOR='vim'" >> ~/.bashrc
#echo "# show the application search bar - note: need to unset and reset this for it to register
#gsettings set org.gnome.shell.keybindings toggle-overview \"[]\"
#gsettings set org.gnome.shell.keybindings toggle-overview \"['<Super>Space']\"" >> ~/.bashrc
echo "# Git branch in prompt.
parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1)/'
}" >> ~/.bashrc
echo "# prompt coloring
export PS1='\[\033[1;31m\]\u@\h\[\033[1;34m\] \w\[\033[1;32m\]\$(parse_git_branch)\[\033[1;36m\] â†’ \[\033[0m\] '" >> ~/.bashrc
echo "# whimsy
fortune | cowsay | lolcat" >> ~/.bashrc

# bash_profile setup
echo "source ~/.bashrc" >> ~/.bash_profile

source ~/.bashrc

vim +'PlugInstall --sync' +qa
