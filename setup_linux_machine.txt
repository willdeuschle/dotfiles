#!/bin/bash -eu

# apt packages
sudo apt update -y
sudo apt install -y vim neovim wget curl gnome-tweak-tool git tmux python3-pip fonts-powerline silversearcher-ag flake8 htop rsync fortune-mod cowsay lolcat xclip
# install chrome
sudo apt install -y libxss1 libappindicator1 libindicator7
wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb
sudo apt install -y ./google-chrome*.deb
rm ./google-chrome*.deb
# install docker
sudo apt install -y apt-transport-https ca-certificates curl gnupg-agent software-properties-common
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
sudo add-apt-repository -y \
   "deb [arch=amd64] https://download.docker.com/linux/ubuntu \
   $(lsb_release -cs) \
   stable"
sudo apt install -y docker-ce docker-ce-cli containerd.io
# create docker group and add user to it
sudo groupadd docker
sudo usermod -aG docker $USER
# get docker-compose
sudo curl -L "https://github.com/docker/compose/releases/download/1.23.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose
# get docker-machine
base=https://github.com/docker/machine/releases/download/v0.16.0 && \
    curl -L $base/docker-machine-$(uname -s)-$(uname -m) >/tmp/docker-machine && \
    sudo install /tmp/docker-machine /usr/local/bin/docker-machine && \
    rm -rf /tmp/docker-machine

# python packages
pip3 install --upgrade neovim

# git setup
if ! [ -f ~/.gitconfig ]; then
    git config --global user.name "Will Deuschle"
    git config --global user.email "wjdeuschle@gmail.com"
fi

# dotfiles setup
if ! [ -d ~/dotfiles ]; then
    git clone https://github.com/willdeuschle/dotfiles.git ~/dotfiles
fi
if ! [ -f ~/.tmux.conf ]; then
    ln -s ~/dotfiles/tmux/.tmux.conf ~/.tmux.conf
fi
if ! [ -f ~/.vimrc ]; then
    ln -s ~/dotfiles/vim/vimrc ~/.vimrc
fi
if ! [ -d ~/.config ]; then
    mkdir ~/.config
fi
if ! [ -d ~/.config/nvim ]; then
    mkdir ~/.config/nvim
fi
if ! [ -f ~/.config/nvim/init.vim ]; then
    ln -s ~/dotfiles/vim/vimrc ~/.config/nvim/init.vim
fi

# setup keyboard shortcuts
source "$PWD/dotfiles/ubuntu_keybindings.txt"

# bashrc
echo "alias vim='nvim'" >> ~/.bashrc
echo "export EDITOR='vim'" >> ~/.bashrc
echo "alias gs='git status'" >> ~/.bashrc
echo "alias gc='git commit'" >> ~/.bashrc
echo "alias gp='git pull'" >> ~/.bashrc
echo "# Git branch in prompt.
parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1)/'
}" >> ~/.bashrc
echo "# prompt coloring
export PS1='\[\033[1;31m\]\u@\h\[\033[1;34m\] \w\[\033[1;32m\]\$(parse_git_branch)\[\033[1;36m\] â†’ \[\033[0m\] '" >> ~/.bashrc
echo "# add (docker) to prompt if in a docker shell
if [ -f /.dockerenv ]; then
    export PS1='(docker) $PS1'
fi" >> ~/.bashrc
echo "# whimsy
fortune | cowsay | lolcat" >> ~/.bashrc

# bash_profile setup
echo "source ~/.bashrc" >> ~/.bash_profile

source ~/.bashrc

vim +'PlugInstall --sync' +qa
